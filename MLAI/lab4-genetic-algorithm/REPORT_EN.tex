\begin{abstract}
This lab applies genetic algorithm to solve the G9 constrained optimization problem. Twenty independent runs validate algorithm performance and parameter sensitivity.
\end{abstract}

\section{Problem Definition}

\subsection{G9 Constrained Optimization}

\textbf{Objective}:
\begin{multline}
\min f(x) = (x_1-10)^2 + 5(x_2-12)^2 + x_3^4 + 3(x_4-11)^2 \\
+ 10x_5^6 + 7x_6^2 + x_7^4 - 4x_6x_7 - 10x_6 - 8x_7
\end{multline}

\textbf{Constraints} ($g_i(x) \leq 0$, $i=1,\ldots,4$)

\textbf{Search space}: $x \in [-10, 10]^7$

\textbf{Global optimum}: $f(x^*) = 680.63$

\section{Methodology}

\subsection{Penalty Function Method}

Transform constrained problem to unconstrained:

\begin{equation}
F(x) = f(x) + \rho \sum_{i=1}^{4} \max(0, g_i(x))
\end{equation}

Penalty factor: $\rho = 10000$
